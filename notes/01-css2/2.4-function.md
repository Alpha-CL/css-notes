# CSS Function


#### calc();

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * calc(express);
 * 
 * @express: 一个数学表达式。结果将用作值。可以使用以下运算符：+ - * /
 * 
 * 允许您执行计算以确定CSS属性值
 */

calc(expression);


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```


####  var();

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * var(custom-property-name, ?value)
 * 
 * 
 * @custom-property-name: 自定义属性的名称（必须以两个破折号开头）;
 * 
 * @value: 后备值（如果自定义属性无效，则使用该值）;
 */


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * set variable
 * 
 * 插入自定义属性的值
 * 
 * --name: value;               // 属性名必须以两个破折号开头
 */

: root {
    --blur: #007bff;
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * get variable
 * 
 * var(--blue);
 */

.demo {
    background-color: var(--blue);
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### cubic-bezier();

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * cubic-bezier();
 * 
 * 定义三次贝塞尔曲线
 */

cubic-bezier(x1,y1,x2,y2);

.demo {
  width: 100px;
  height: 100px;
  background: red;
  transition: width 2s;
  transition-timing-function: cubic-bezier(0.1, 0.7, 1.0, 0.1);
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### attr();

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * attr(attribute-name);
 * 
 * @attribute-name: HTML元素的 '属性名称';
 * 
 * 
 * 返回所选元素指定属性的属性值
 */

attr(attribute-name);

eg: 

a:after {
  content: " (" attr(href) ")";             // 返回当前元素的 href 的属性值
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
