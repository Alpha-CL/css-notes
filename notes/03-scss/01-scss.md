### @extend

#### extend properties

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 继承
 * 
 * 
 * 将另一个选择器的样式继承过来( 类似于 js 中的 class )
 * 
 ** 仅可继承 单个选择器( p {}, .demo {} )，不可继承 选择器列( .foo .bar {} )
 */


.class {

    border: 1px solid;
    color: #fdd;
}

.extend {

    @extend: .class;                // 将 .class 中所有的样式继承过来
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 继承可来自于多个类


.parent_01 {

    font-size: 12px;
}

.parent_02 {

    color: red;
}

.children {

    @extend: .parent_01;
    @extend: .parent_02;
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 多继承链


.grandFather {

    font-size: 12px;
}


.father {

    color: red;
    @extend: .grandFather;
}


.son {

    @extend: .grandFather;
    @extend: father;
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 仅可继承自 单个选择器，无法继承自 选择器列


.a .b .c {

}


.d {

    @extend: .c;                // 仅可继承自 单个 选择器

    @extend: .a .b .c;          // 无法继承自 选择器列
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * 阵列选择器继承 的编译过程 是智能的
 * 
 * 
 */


// 当两个列 (sequence) 合并时，如果没有包含相同的选择器，将生成两个新选择器：第一列出现在第二列之前，或者第二列出现在第一列之前

.a .b .c {

    color: red;
}

.d .e .f {

    @extend .c;
}

// result

.a .b .c,
.a .b .d .e .f,
.d .e .a .b .f {
  color: red;
}


// 如果两个列 (sequence) 包含了相同的选择器，相同部分将会合并在一起，其他部分交替输出


.a .b .c {

    color: red;
}

.a .d .e {

    @extend .c;
}

// result

.a .b .c, 
.a .b .d .e,
.a .d .b .e {
    color: red;
}
```

#### argument

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * mixin & include 可以理解 js 的函数
 * 
 * 
 * 1) 可以像函数一样使用 参数( 必须以 $ 开头 )
 * 
 *      - 单个参数      mixin($param){}
 *      - 多个参数      mixin($p1,$p2,$p3){}
 *      - 剩余参数      mixin($params...){}
 * 
 * 2) 可以给参数设置 '默认值'
 * 
 */


@mixin [key]($[param]: defualtVal) {

    // @$[param]: mixin 可传入一个或多个参数，也可传入变量作为参数
    // @defualtVal: 对应指定参数的默认值
}

.demo {

    @include: [mixinKey]([param: $variable | any]);
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// mixin 函数中可以使用关键字 作为参数 eg: color, width ...

p { @include sexy-border($color: blue); }

h1 { @include sexy-border($color: blue, $width: 2in); }


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
