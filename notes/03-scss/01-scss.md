## extend/inheritance

#### extend properties

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 继承
 * 
 * 
 * 将另一个选择器的样式继承过来( 类似于 js 中的 class )
 * 
 ** 仅可继承 单个选择器( .demo {} )，不可继承 选择器列( .foo .bar {} )
 */


.class {

    border: 1px solid;
    color: #fdd;
}


.extend {

    @extend: .class;                // 将 .class 中所有的样式继承过来
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 继承可来自于多个类


.parent_01 {

    font-size: 12px;
}

.parent_02 {

    color: red;
}

.children {

    @extend: .parent_01;
    @extend: .parent_02;
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 多继承链


.grandFather {

    font-size: 12px;
}


.father {

    color: red;
    @extend: .grandFather;
}


.son {

    @extend: .grandFather;
    @extend: father;
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


// 仅可继承自 单个选择器，无法继承自 选择器列


.a .b .c {

}


.d {

    @extend: .c;                // 仅可继承自 单个 选择器

    @extend: .a .b .c;          // 无法继承自 选择器列
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * 阵列选择器继承 的编译过程 是智能的
 * 
 * 
 */


// 阵列选择器继承中，未出现相同的选择器

.a .b .c {

    color: red;
}

.d .e .f {

    @extend .c;
}

// result

.a .b .c,
.a .b .d .e .f,
.d .e .a .b .f {
  color: red;
}


// 阵列选择器继承中，出现相同的选择器


.a .b .c {

    color: red;
}

.a .d .e {

    @extend .c;
}

// result

.a .b .c, 
.a .b .d .e,
.a .d .b .e {
    color: red;
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### mathematical operator

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * + - * /
 * 
 * 
 * 加，减，乘，除
 */


/**
 * ()
 * 
 * 
 * 提升运算优先级
 */


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * /            // 分隔符 & 除法
 * 
 * 
 * 以下三种情况作为 数学运算判断
 */


.demo {

  width: $width/2;                  // 1. 如果值，或值的一部分，是变量或者函数的返回值 
  width: round(1.5)/2;              // 1. 如果值，或值的一部分，是变量或者函数的返回值
  
  height: (500px/2);                // 2. 如果值被圆括号包裹
      
  margin-left: 5px + 8px/2px;       // 3. 如果值是算数表达式的一部分
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * +            // 加法 & 字符串连接
 * 
 * 
 * 字符串在 + 左侧时返回 "有引号字符串"，反之为 "无引号字符串"
 */


1) 字符串 在 + 左侧时返回 "有引号字符串"

.demo {

    content: 'foo' + bar;               // 'foobar'
}


2) 字符串 在 + 左侧时返回  无引号字符串


.demo {

    content: foo + 'bar';               // foobar
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
